338  private async Task<bool> CheckForUpdatesHttps()
339  {
340  _hasCheckedForUpdates = true;
341  if (await UpdateChecker.CheckForUpdates(VersionNumber, ReleaseType.Equals(ERelease.PRERELEASE)).ConfigureAwait(false) is (SemVersion, GithubReleaseHttpResponse) update)
342  {
343  (SemVersion newVersion, GithubReleaseHttpResponse response) = update;
344  switch (MessageBox.Show(
345  $"Current Version: {VersionNumber}\n" +
346  $"Newest Version: {newVersion}\n" +
347  "\nDo you want to update now?\n\nClick 'Yes' to update now.\nClick 'No' to update later.\nClick 'Cancel' to disable update checking."
348  , "Update Available", MessageBoxButton.YesNoCancel, MessageBoxImage.Question, MessageBoxResult.No))
349  {
350  case MessageBoxResult.Yes: // open browser
351  string asset_url = response.assets.First(a => a.name.Equals("VolumeControl.exe", StringComparison.OrdinalIgnoreCase)).browser_download_url;
352  Log.Info($"Updating to version {newVersion}: '{asset_url}'");
353 
354  if (SetupUpdateUtility() is string path)
355  {
356  Log.Info($"Automatic update utility was created at {path}");
357  ProcessStartInfo psi = new(path, $"--url \"{asset_url}\" --path {ExecutablePath}")
358  {
359  ErrorDialog = true,
360  UseShellExecute = true,
361  };
362  Process.Start(psi);
363  return true;
364  }
365  else
366  {
367  Log.Error("Failed to setup automatic update utility, falling back to the default web browser.");
368  UpdateChecker.OpenBrowser(response.html_url);
369  }
370  break;
371  case MessageBoxResult.No: // do nothing
372  Log.Info($"Version {newVersion} is available, but was temporarily ignored.");
373  break;
374  case MessageBoxResult.Cancel: // disable
375  Settings.CheckForUpdatesOnStartup = false;
376  Log.Info("Disabled automatic updates");
377  Settings.Save();
378  Settings.Reload();
379  break;
380  }
381  }
382  return false;
383  }
389  internal string? SetupUpdateUtility(int asyncTimeout = -1)
390  {
391  var asm = Assembly.GetEntryAssembly();
392  if (asm == null)
393  {
394  return null;
395  }
396 
397  string path = Path.Combine(Path.GetDirectoryName(ExecutablePath) ?? string.Empty, _updateUtilityFilename);
398 
399  if (File.Exists(path)) // if the file already exists, delete it
400  {
401  Log.Warning($"Deleted update utility from a previous update located at '{path}'");
402  File.Delete(path);
403  }
404 
405  Stream? s = null;
406 
407  try
408  {
409  s = asm.GetManifestResourceStream(_updateUtilityResourcePath);
410  }
411  catch (Exception ex)
412  {
413  Log.Error(
414  "An exception was thrown while creating the update utility!",
415  $"{{ Resource: '{_updateUtilityResourcePath}' }}",
416  ex);
417  }
418  if (s == null)
419  {
420  Log.Error($"Failed to get a stream containing resource '{_updateUtilityResourcePath}'!", "This indicates that something may have gone wrong during the build process!", "Please attach a copy of this log file and report this at https://github.com/radj307/volume-control/issues");
421  return null;
422  }
423  else if (s.Length <= 0)
424  {
425  Log.Error("Failed to retrieve the embedded update utility resource file!", "This indicates that something may have gone wrong during the build process!", "Please attach a copy of this log file and report this at https://github.com/radj307/volume-control/issues");
426  s.Close();
427  s.Dispose();
428  return null;
429  }
430 
431  using Stream fs = File.Open(path, FileMode.OpenOrCreate, FileAccess.Write, FileShare.Read);
432  fs.SetLength(s.Length);
433  s.CopyTo(fs);
434  s.Close();
435  s.Dispose();
436  s = null;
437 
438  fs.Flush(); //< idk if this is required, but it's probably a good idea
439  fs.Close();
440  fs.Dispose();
441 
442  if (!File.Exists(path))
443  {
444  return null;
445  }
446 
447  return path;
448  }
