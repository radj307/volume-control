48  public void LoadAddons(ref List<IBaseAddon> addons)
49  {
50  if (AddonDirectories.Count == 0) return;
51  Log.Debug($"Searching for addon assemblies in {AddonDirectories.Count} director{(AddonDirectories.Count == 1 ? "y" : "ies")}.");
52  int asmCount = 0,
53  totalCount = 0;
54  foreach (string dir in AddonDirectories)
55  {
56  if (!Directory.Exists(dir))
57  {
58  Log.Warning($"Addon directory '{dir}' doesn't exist.");
59  continue;
60  }
61 
62  Log.Debug($"Searching for Addon Assemblies in '{dir}'");
63 
64  foreach (string path in Directory.EnumerateFiles(dir, "*.dll", DirectoryEnumerationOptions))
65  {
66  string prefix = $" [{++totalCount}] ";
67  var asm = Assembly.LoadFrom(path);
68 
69  int typeCount = LoadAddonTypes(ref addons, asm, _currentVersion);
70 
71  if (typeCount == 0)
72  {
73  Log.Debug($"{prefix}No classes found in assembly '{asm.FullName}'");
74  continue;
75  }
76  Log.Debug($"{prefix}Loaded {typeCount} classes from assembly '{asm.FullName}'.");
77  prefix = new(' ', prefix.Length);
78 
79  ++asmCount;
80  Log.Debug($"{prefix}Cached addon '{asm.FullName}'");
81  }
82  }
83  Log.Debug($"Loaded {asmCount} out of {totalCount} assemblies from {AddonDirectories.Count} director{(AddonDirectories.Count == 1 ? "y" : "ies")}");
84  // initialize addons
85  foreach (var addon in addons)
86  {
87  addon.LoadFromTypes();
88  Log.Debug($"Initialized {addon.Attribute.Name}");
89  }
90  }
