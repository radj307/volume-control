558  private List<AudioSession> GetAllSessionsFromAllDevices()
559  {
560  List<AudioSession> sessions = new();
561 
562  for (int i = Devices.Count - 1; i >= 0; --i)
563  {
564  if (Devices[i] is AudioDevice dev)
565  {
566  switch (dev.State)
567  {
568  case DeviceState.NotPresent:
569  case DeviceState.Disabled:
570  case DeviceState.Unplugged:
571  case DeviceState.All:
572  continue; // skip devices that aren't active
573  case DeviceState.Active:
574  default:
575  break; // proceed if the device is active or unknown
576  }
577 
578  // add non-duplicate sessions to the list
579  sessions.AddRange(dev.GetAudioSessions().Where(devSession => !sessions.Any(s => s.ProcessIdentifier.Equals(devSession.ProcessIdentifier, StringComparison.Ordinal))));
580  }
581  else
582  {
583  Log.Error($"{nameof(Devices)}[{i}] contains invalid type '{Devices[i].GetType()}'!");
584  }
585  }
586 
587  return sessions;
