78  {
79  if (!Endpoint.Enabled || lines.Length == 0)
80  return;
81  using TextWriter w = Endpoint.GetWriter()!;
82  w.Write(ts);
83  string tsBlank = MakeBlankTimestamp();
84  for (int i = 0, end = lines.Length; i < end; ++i)
85  {
86  object? line = lines[i];
87 
88  if (line is null)
89  {
90  continue;
91  }
92  else if (line is Exception ex)
93  {
94  w.WriteLine($"{(i == 0 ? "" : tsBlank)}{ex.ToString(tsBlank.Length)}");
95  }
96  else if (line is string s)
97  {
98  if (s.Length > 0)
99  w.WriteLine($"{Regex.Replace(s, "\\B\\r*\\n", $"\n{tsBlank}")}");
100  }
101  else if (line is IEnumerable enumerable)
102  {
103  foreach (object? item in enumerable)
104  {
105  if (item != null)
106  w.WriteLine($"{(i == 0 ? "" : tsBlank)}{item}");
107  }
108  }
109  else
110  {
111  w.WriteLine($"{(i == 0 ? "" : tsBlank)}{line}");
112  }
113  }

116  }
