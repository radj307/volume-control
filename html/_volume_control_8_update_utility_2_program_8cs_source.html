109 
110  if (alreadyExists)

112  if (File.Exists(backupPath))
113  File.Delete(backupPath);
114  File.Move(args.ExecutablePath, backupPath!);
115  Log.Info($"Created backup: '{args.ExecutablePath}' => '{backupPath}'");
116  }

120  // Set the request header to request the raw executable file
121  client.DefaultRequestHeaders.Accept.Clear();

124

132  File.Delete(backupPath);
133  Log.Info($"Deleted backup '{backupPath}'");
134  }
135  catch (Exception ex)

138  }
139  }

141  Log.Info("Successfully updated Volume Control.");
142  }
143  else
144  { // timeout

147  {
148  File.Move(backupPath!, args.ExecutablePath);

150  }
