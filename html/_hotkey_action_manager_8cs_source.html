100  protected virtual void MergeActionBindingsList(List<IActionBinding> list, NameConflictResolution resolutionType = NameConflictResolution.Throw)
101  {
102  foreach (IActionBinding? action in list)
103  {
104  if (Bindings.Any(b => b.Name.Equals(action.Name, StringComparison.OrdinalIgnoreCase)))
105  {
106  switch (resolutionType)
107  {
108  case NameConflictResolution.Throw:
109  throw new Exception($"An action binding with the name '{action.Name}' already exists!");
110  case NameConflictResolution.Ignore:
111  continue; //< continue to next element
112  case NameConflictResolution.Overwrite:
113  break; //< break to add action
114  }
115  }
116  NotifyPropertyChanging(nameof(Bindings));
